CREATE TABLE DIRECCION(
IDDIRECCION NUMBER(10),
ID_CLIENTE NUMBER(25),
PROVINCIA NUMBER(25),
CANTON NUMBER(25),
DISTRITO  NUMBER(25),
OTRAS_SENAS VARCHAR2(25)
);

ALTER TABLE DIRECCION ADD CONSTRAINT PK_IDDIRECCION PRIMARY KEY(IDDIRECCION);

ALTER TABLE DIRECCION ADD CONSTRAINT FK_IDCliente 
FOREIGN KEY(ID_CLIENTE) REFERENCES Cliente(IDCliente);

ALTER TABLE DIRECCION ADD CONSTRAINT FK_ID_PROVINCIA 
FOREIGN KEY(PROVINCIA) REFERENCES PROVINCIA(ID_PROVINCIA);

ALTER TABLE DIRECCION ADD CONSTRAINT FK_ID_CANTON 
FOREIGN KEY(CANTON) REFERENCES CANTON(ID_CANTON);

ALTER TABLE DIRECCION ADD CONSTRAINT FK_ID_DISTRITO

CREATE TABLE FACTURA(
ID_FACTURA VARCHAR2(10),
FECHA DATE,
ID_CLIENTE NUMBER(25),
FERRETERIA NUMBER (25)
);
ALTER TABLE FACTURA ADD CONSTRAINT PK_ID_FACTURA PRIMARY KEY(ID_FACTURA)

ALTER TABLE FACTURA ADD CONSTRAINT FK_CLIENTE
FOREIGN KEY(ID_CLIENTE) REFERENCES Cliente(IDCliente);

CREATE TABLE FERRETERIA(
CODIGO NUMBER(25) CONSTRAINT PK_FERRETERIA  PRIMARY KEY,
FERRETERIA VARCHAR2(20)
);

CREATE TABLE ARTICULOS(
IDARTICULO NUMBER(10) CONSTRAINT PK_ARTICULO  PRIMARY KEY,
DESCRIPCION VARCHAR2(25),
EXISTENCIA NUMBER(25),
PRECIO_UNITARIO NUMBER(20),
DESCUENTO  NUMBER(20)
);


CREATE TABLE DETALLE_ARTICULO(
IDDETALLE NUMBER(10) CONSTRAINT PK_DETALLE_ARTICULO  PRIMARY KEY,
FACTURA VARCHAR2(25),
ID_ARTICULO NUMBER(25),
CANTIDAD NUMBER(20),
PRECIO_UNITARIO NUMBER(20),
DESCUENTO NUMBER(20)
);

ALTER TABLE DETALLE_ARTICULO ADD CONSTRAINT FK_DETALLE_FACTURA
FOREIGN KEY(ID_ARTICULO) REFERENCES ARTICULOS(IDARTICULO);

ALTER TABLE DETALLE_ARTICULO ADD CONSTRAINT FK_DETALLE_FACTURA1
FOREIGN KEY(FACTURA) REFERENCES FACTURA(ID_FACTURA);

CREATE TABLE PROVINCIA(
ID_PROVINCIA NUMBER(10),
NOMBRE_PROVINCIA VARCHAR2(25)
);
ALTER TABLE PROVINCIA ADD CONSTRAINT PK_PROVINCIA PRIMARY KEY(ID_PROVINCIA)
CREATE TABLE CANTON(
ID_CANTON NUMBER(10),
NOMBRE_CANTON VARCHAR2(25)
);
ALTER TABLE CANTON ADD CONSTRAINT PK_CANTON PRIMARY KEY(ID_CANTON)

CREATE TABLE DISTRITO(
ID_DISTRITO NUMBER(10),
NOMBRE_DISTRITO VARCHAR2(25)
);
ALTER TABLE DISTRITO ADD CONSTRAINT PK_DISTRITO PRIMARY KEY(ID_DISTRITO)
